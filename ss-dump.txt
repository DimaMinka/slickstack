####################################################################################################
#### SS-Dump: Export The Live MySQL Database #######################################################
####################################################################################################

## MESSAGE ##
echo -e "\e[36mDumping the live MySQL database to /var/www/wp.sql for safe-keeping...\e[0m"

## prepare password (suppresses warning messages) ##
# export MYSQL_PWD=$DB_PASSWORD_ROOT

## dump MySQL database to /var/www/wp.sql ##
mysqldump --user=root --password="$DB_PASSWORD_ROOT" --host="$DB_HOST" --protocol=tcp --port=3306 --dump-date --flush-privileges --force "$DB_NAME" > /var/www/wp.sql

####################################################################################################
#### SS-Dump: Reset Permissions (Allows SFTP User To Access The SQL Dump File) #####################
####################################################################################################

chown -R $SFTP_USER:www-data /var/www/wp.sql
chmod 6775 /var/www/wp.sql
