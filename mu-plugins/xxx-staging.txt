/** Staging Admin Notices */
add_action('admin_notices', 'staging_admin_notices');

function staging_admin_notices() {
    $current_user = wp_get_current_user();
    if(current_user_can('manage_options')) { ?>

    <div class="notice notice-warning">
    <p><strong>FOR STAGING PURPOSES ONLY:</strong> You are now browsing a temporary staging site, created automagically from your live production site. Data is continuously synced every 30 minutes and will overwrite existing data (i.e. all staging files/database). Both sites share the <code>/uploads/</code> directory from the production site to avoid confusion and conflicts, so be sure not to upload any media files to this staging site. Please use this site to briefly test new plugins, themes, or features (and not for long-term web design work). To avoid conflicts and database bloat, it is also best to not push your staging database to your production (live) site, and instead use this site for testing purposes only. However, if you do push your staging site live, keep in mind that only <strong>Plugins</strong>, <strong>Themes</strong>, and the <code>wp_options</code> table will be synced, and nothing else (this process will overwrite the Plugins, Themes, and <code>wp_options</code> table on your production site). If one of your plugins or themes is improperly using the <code>/uploads/</code> directory (or otherwise) to store critical files, THESE FILES WILL BE IGNORED WHEN PUSHING FROM STAGING TO PRODUCTION.</p>
    </div>

    <?php } else { }
}

/** Toolbar Switcher (Staging) */
add_action('admin_bar_menu', 'slickstack_switcher_staging', 25);

function slickstack_switcher_staging() {
    $current_user = wp_get_current_user();
    if(current_user_can('manage_options')) {
        global $wp_admin_bar;
        $menu_id = 'slickstack_switcher_staging';
        $prod_link = home_url( $path = '', $scheme = https );
        $wp_admin_bar->add_menu(array('id' => $menu_id, 'title' => __('Staging'), 'href' => '/'));
        $wp_admin_bar->add_menu(array('parent' => $menu_id, 'title' => __('Switch To Production'), 'id' => 'production-link', 'href' => '$prod_link'));
    } else { }
}
